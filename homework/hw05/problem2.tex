\item * Consider the following joint distribution for two ternary random variables:
\[  \begin{array}{r|ccc}
& \multicolumn{3}{c}{X} \\
Y & 0 & 1 & 2 \\
\hline
0 & 0 & 0.2 & 0 \\
1 & 0.1 & 0.1 & 0.1 \\
2 & 0.2 & 0.1 & 0.2 \\
\hline
\end{array} \]
\begin{enumerate}
\item Find the marginal probability mass functions for $X$ and $Y$.
\item Are $X$ and $Y$ independent?
\item Calculate the expectation and variance for $X$ and $Y$.
\item Calculate the covariance and correlation between $X$ and $Y$.
\end{enumerate}

\ansfont{
\begin{enumerate}
\item The marginal distributions for X and Y are 

\[ 
\begin{array}{c|ccc}
x & 0 & 1 & 2 \\
\hline
p_X(x) & 0.3 & 0.4 & 0.3 \\
\hline
\end{array}  
%
\qquad
%
\begin{array}{c|ccc}
y & 0 & 1 & 2 \\
\hline
p_Y(y) & 0.2 & 0.3 & 0.5 \\
\hline
\end{array} 
\] 

\item No, $X$ and $Y$ are not independent since 
\[ 0 = P(X=0,Y=0) \ne P(X=0)P(Y=0) = 0.3\times 0.2 = 0.06 \] 

\item The expectation and variance are 
\begin{align*}
E[X] &= 0.3\times 0 + 0.4 \times 1 + 0.3 \times 2 = 1 \\
V[X] &= 0.3\times (0-1)^2 + 0.4\times (1-1)^2 + 0.3\times( 2-1)^2 = 0.6 \\
E[Y] &= 0.2 \times 0 + 0.3\times 1 + 0.5\times 2 = 1.3 \\
V[Y] &= 0.2\times (0-1.3)^2 + 0.3\times (1-1.3)^2 + 0.5\times (2-1.3)^2 = 0.61
\end{align*}

\item The covariance and correlation are  
\begin{align*}
Cov(X,Y) =& E[(X-E[X])(Y-E[Y]) \\
=& 0 \times (0-1)(0-1.3) + 0.2 \times (1-1)(0-1.3) + 0\times (2-1)(0-1.3) \\
&+ 0.1\times (0-1)(1-1.3) + 0.1\times (1-1)(1-1.3) + 0.1\times (2-1)(1-1.3) \\
&+ 0.2\times (0-1)(2-1.3)+0.1\times(1-1)(2-1.3)+0.2\times(2-1)(2-1.3) \\
&= 0.03 - 0.03 - 0.14 + 0. 14 \\
&= 0 \\
Corr(X,Y) =& \rho = \frac{Cov(X,Y)}{\sqrt{V[X]V[Y]}} \\
=& \frac{0}{\sqrt{0.6\times 0.569}} = 0
\end{align*}
This is an example of two random variables that are not independent but the correlation is zero. 

\end{enumerate}
